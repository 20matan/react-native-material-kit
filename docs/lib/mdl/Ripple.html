      _onTouchEvent(evt) {

      case 'TOUCH_DOWN':
        this._onPointerDown(evt);
        break;
      case 'TOUCH_UP':
      case 'TOUCH_CANCEL':
        this._onPointerUp();
        break;

  _onLayout(evt) {
    this._onLayoutChange(evt.nativeEvent.layout);

    if (this.props.onLayout) {
      this.props.onLayout(evt);
    }
  }

  _onLayoutChange({width, height}) {
    if (width === this.state.width && height === this.state.height) {
      return;

    this.setState({
      width,
      height,
