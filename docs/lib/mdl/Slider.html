    class Slider extends Component {
  constructor(props) {
    super(props);
    this._panResponder = PanResponder.create({
      onStartShouldSetPanResponder: (evt, gestureState) => true,
      onStartShouldSetPanResponderCapture: (evt, gestureState) => true,
      onMoveShouldSetPanResponder: (evt, gestureState) => true,
      onMoveShouldSetPanResponderCapture: (evt, gestureState) => true,
      onPanResponderGrant: (evt, gestureState) => {
        const e = this._getTouchEvent("TOUCH_DOWN", gestureState);
        this._onTouchEvent(e);
      },
      onPanResponderMove: (evt, gestureState) => {
        const e = this._getTouchEvent("TOUCH_MOVE", gestureState);
        this._onTouchEvent(e);
      },
      onPanResponderTerminationRequest: (evt, gestureState) => true,
      onPanResponderRelease: (evt, gestureState) => {
        const e = this._getTouchEvent("TOUCH_UP", gestureState);
        this._onTouchEvent(e);
      },
      onPanResponderTerminate: (evt, gestureState) => {
        const e = this._getTouchEvent("TOUCH_CANCEL", gestureState);
        this._onTouchEvent(e);
      },
      onShouldBlockNativeResponder: (evt, gestureState) => true,
    });
    this._value = 0;
    this._trackTotalLength = 0;
    this._pageX = 0;
    this._animatedTrackLength = new Animated.Value(0);
  }

  componentWillMount() {
    this._onThumbRadiiUpdate(this.props);

  componentWillReceiveProps(nextProps) {
    this._onThumbRadiiUpdate(nextProps);

  _internalSetValue(value, fireChange=true) {
    this._value = value;
    if (fireChange) {
      this._emitChange(value);
    }
  }

  _emitChange(newValue) {
    if (this.props.onChange) {
      this.props.onChange(newValue);
    }
  }

  _aniUpdateValue(value) {
    if (!this._trackTotalLength) {
      return;
    }

    const ratio = (value - this.props.min) / (this.props.max - this.props.min);
    const x = ratio * this._trackTotalLength;
    this._moveThumb(x);
    this._confirmMoveThumb(x);
  }

  _getTouchEvent(type, gestureState) {
    const x = gestureState.x0 - this._pageX + gestureState.dx;
    console.log('touch x0:%d dx:%d, _pageX:%d', gestureState.x0, gestureState.dx, this._pageX);
    return { type: type, x: x, y: 0 };
  }
